#include <ctype.h>

void to_upper(char* s) {
    while (*s) {
        *s = toupper((unsigned char)*s);
        s++;

void parse_input(char* input, char* verb, char* noun) {
    // Convert to uppercase for case-insensitive matching
    to_upper(input);

    // Tokenize into verb and noun
    char* token = strtok(input, " \t\n");
    if (token) {
        strcpy(verb, token);
        token = strtok(NULL, " \t\n");
        if (token) {
            strcpy(noun, token);
        } else {
            noun[0] = '\0';
        }
    } else {
        verb[0] = '\0';
        noun[0] = '\0';
    }
}

3. Match the verb to a command:

CommandID lookup_command(const char* verb) {
    for (int i = 0; i < NUM_COMMANDS; i++) {
        if (strcmp(verb, command_table[i]) == 0) {
            return (CommandID)i;
        }
    }
    return CMD_UNKNOWN;
}

4. Handle the command:

void execute_command(CommandID cmd, const char* noun) {
    switch (cmd) {
        case CMD_GO:
            printf("You try to go %s.\n", noun);
            break;
        case CMD_LOOK:
            printf("You look around.\n");
            break;
        case CMD_TAKE:
            printf("You try to take %s.\n", noun);
            break;

            break;
        case CMD_QUIT:

            break;
    }

int main() {

    printf("Welcome to the Adventure!\n");

        parse_input(input, verb, noun);
        CommandID cmd = lookup_command(verb);
        execute_command(cmd, noun);

        if (cmd == CMD_QUIT)
            break;
    }

    return 0;
}
