Nice! Using tables and enums for verbs and nouns in a text adventure parser is a classic and clean approach, especially for low-power environments and C.

Here’s a simple, extendable pattern you could follow — it keeps your verbs, nouns, and their possible interactions separate from your main code, so adding new actions or objects doesn’t mean rewriting big chunks of logic.

Step 1: Define enums for verbs and nouns

c
typedef enum {
    VERB_LOOK,
    VERB_TAKE,
    VERB_USE,
    VERB_OPEN,
    VERB_CLOSE,
    VERB_INVENTORY,
    VERB_UNKNOWN
} Verb;

typedef enum {
    NOUN_DOOR,
    NOUN_KEY,
    NOUN_ROOM,
    NOUN_CHEST,
    NOUN_UNKNOWN
} Noun;

Step 2: Define string tables for easy mapping from user input

c
typedef struct {
    const char *name;
    Verb verb;
} VerbEntry;

static const VerbEntry verbs[] = {
    {"look", VERB_LOOK},
    {"take", VERB_TAKE},
    {"use", VERB_USE},
    {"open", VERB_OPEN},
    {"close", VERB_CLOSE},
    {"inventory", VERB_INVENTORY},
    {NULL, VERB_UNKNOWN}  // terminator
};

typedef struct {
    const char *name;
    Noun noun;
} NounEntry;

static const NounEntry nouns[] = {
    {"door", NOUN_DOOR},
    {"key", NOUN_KEY},
    {"room", NOUN_ROOM},
    {"chest", NOUN_CHEST},
    {NULL, NOUN_UNKNOWN}  // terminator
};

Step 3: Parsing helpers to convert strings to enums

c
Verb parse_verb(const char *input) {
    for (int i = 0; verbs[i].name != NULL; i++) {
        if (strcmp(input, verbs[i].name) == 0) {
            return verbs[i].verb;
        }
    }
    return VERB_UNKNOWN;
}

Noun parse_noun(const char *input) {
    for (int i = 0; nouns[i].name != NULL; i++) {
        if (strcmp(input, nouns[i].name) == 0) {
            return nouns[i].noun;
        }
    }
    return NOUN_UNKNOWN;
}

Step 4: Define a command table linking verb-noun combos to handler functions

c
typedef void (*CommandHandler)(void);

typedef struct {
    Verb verb;
    Noun noun;
    CommandHandler handler;
} CommandEntry;

Step 5: Example handlers and command table

c
void handle_look_door(void) {
    printf("The door is old and creaky.\n");
}

void handle_take_key(void) {
    printf("You pick up the rusty key.\n");
}

void handle_unknown(void) {
    printf("You can't do that.\n");
}

static const CommandEntry commands[] = {
    {VERB_LOOK, NOUN_DOOR, handle_look_door},
    {VERB_TAKE, NOUN_KEY, handle_take_key},
    // Add more combinations here
    {VERB_UNKNOWN, NOUN_UNKNOWN, handle_unknown}  // fallback
};

Step 6: Running a command

c
void run_command(Verb v, Noun n) {
    for (int i = 0; i < sizeof(commands)/sizeof(commands[0]); i++) {
        if (commands[i].verb == v && commands[i].noun == n) {
            commands[i].handler();
            return;
        }
    }
    handle_unknown();
}

How to use

c
int main() {
    char verb_str[20], noun_str[20];
    printf("> ");
    scanf("%19s %19s", verb_str, noun_str);

    Verb v = parse_verb(verb_str);
    Noun n = parse_noun(noun_str);

    run_command(v, n);

    return 0;
}

Advantages

  • You only edit the verbs[], nouns[], and commands[] tables to add or modify your game’s vocabulary or behaviors.

  • Clear separation of parsing, data, and game logic.

  • Easy to scale and optimize.

✅ Complete Example

c
#include <stdio.h>
#include <string.h>

// === Verb system ===

typedef enum {
    VERB_LOOK,
    VERB_TAKE,
    VERB_OPEN,
    VERB_UNKNOWN
} Verb;

const char *verb_names[] = {
    "look",
    "take",
    "open"
};

// === Noun system ===

typedef enum {
    NOUN_DOOR,
    NOUN_KEY,
    NOUN_CHEST,
    NOUN_ROOM,
    NOUN_UNKNOWN = -1
} Noun;

typedef enum {
    ATTR_NONE      = 0,
    ATTR_CONTAINER = 1 << 0,
    ATTR_LOCKED    = 1 << 1,
    ATTR_PICKABLE  = 1 << 2
} NounAttributes;

typedef struct {
    const char *name;
    const char *description;
    NounAttributes attrs;
} NounData;

const NounData noun_data[] = {
    {"door",  "An old wooden door with rusty hinges.", ATTR_CONTAINER | ATTR_LOCKED},
    {"key",   "A small rusty key.", ATTR_PICKABLE},
    {"chest", "A large wooden chest bound with iron.", ATTR_CONTAINER | ATTR_LOCKED},
    {"room",  "You are in a dimly lit stone room.", ATTR_NONE}
};

const int NOUN_COUNT = sizeof(noun_data) / sizeof(noun_data[0]);

// === Parser functions ===

Verb parse_verb(const char *word) {
    for (int i = 0; i < VERB_UNKNOWN; i++) {
        if (strcmp(word, verb_names[i]) == 0) {
            return (Verb)i;
        }
    }
    return VERB_UNKNOWN;
}

Noun parse_noun(const char *word) {
    for (int i = 0; i < NOUN_COUNT; i++) {
        if (strcmp(word, noun_data[i].name) == 0) {
            return (Noun)i;
        }
    }
    return NOUN_UNKNOWN;
}

// === Verb handlers ===

void handle_look(Noun noun) {
    if (noun == NOUN_UNKNOWN) {
        printf("Look at what?\n");
        return;
    }
    printf("%s\n", noun_data[noun].description);
}

void handle_take(Noun noun) {
    if (noun == NOUN_UNKNOWN) {
        printf("Take what?\n");
        return;
    }
    if (noun_data[noun].attrs & ATTR_PICKABLE) {
        printf("You pick up the %s.\n", noun_data[noun].name);
    } else {
        printf("You can't take the %s.\n", noun_data[noun].name);
    }
}

void handle_open(Noun noun) {
    if (noun == NOUN_UNKNOWN) {
        printf("Open what?\n");
        return;
    }
    if (!(noun_data[noun].attrs & ATTR_CONTAINER)) {
        printf("You can't open the %s.\n", noun_data[noun].name);
    } else if (noun_data[noun].attrs & ATTR_LOCKED) {
        printf("The %s is locked.\n", noun_data[noun].name);
    } else {
        printf("You open the %s.\n", noun_data[noun].name);
    }
}

// === Command table ===

typedef void (*VerbHandler)(Noun noun);

typedef struct {
    Verb verb;
    VerbHandler handler;
} CommandEntry;

const CommandEntry command_table[] = {
    {VERB_LOOK, handle_look},
    {VERB_TAKE, handle_take},
    {VERB_OPEN, handle_open},
    {VERB_UNKNOWN, NULL}
};

// === Command runner ===

void run_command(Verb verb, Noun noun) {
    for (int i = 0; command_table[i].verb != VERB_UNKNOWN; i++) {
        if (command_table[i].verb == verb) {
            command_table[i].handler(noun);
            return;
        }
    }
    printf("You can't do that.\n");
}

// === Input tokenizer ===

void parse_and_execute(const char *input) {
    char verb_word[16], noun_word[16];
    if (sscanf(input, "%15s %15s", verb_word, noun_word) < 1) {
        printf("I didn't understand that.\n");
        return;
    }

    Verb verb = parse_verb(verb_word);
    Noun noun = parse_noun(noun_word);

    run_command(verb, noun);
}

// === Main loop ===

int main(void) {
    char input[64];
    printf("Text Adventure Ready.\n");

    while (1) {
        printf("> ");
        if (!fgets(input, sizeof(input), stdin)) break;
        
        // Strip newline if present
        size_t len = strlen(input);
        if (len > 0 && input[len - 1] == '\n')
            input[len - 1] = '\0';

        parse_and_execute(input);
    }

    return 0;
}
